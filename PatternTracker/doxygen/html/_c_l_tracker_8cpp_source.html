526 bool CLTracker::getPatternIdAndIntensity(cv::Mat &locMat, cv::Mat &frameMetadataI,
527  cv::Mat &frameMetadataF, int patId_prev, int debug) {
528  cl_int err, ret;
529  // getting the indicator bits
530  float endPts[12][2];
531  float k = 30.0 / 23.0; // it is point 7/8th inter point length further from end point
532  for (int i = 0; i < 3; i++) {
533  // horizontal lines
534  float pt0x = locMat.at<float>(i, 0);
535  float pt0y = locMat.at<float>(i, 1);
536  float pt1x = locMat.at<float>(3 + i, 0);
537  float pt1y = locMat.at<float>(3 + i, 1);
538  float pt2x = locMat.at<float>(6 + i, 0);
539  float pt2y = locMat.at<float>(6 + i, 1);
540 
541  float a = std::sqrt(
542  (pt0x - pt1x) * (pt0x - pt1x) + (pt0y - pt1y) * (pt0y - pt1y));
543  float b = std::sqrt(
544  (pt2x - pt1x) * (pt2x - pt1x) + (pt2y - pt1y) * (pt2y - pt1y));
545  float c0 = a * (b + a) * (k - 1) / (b + a * (1 - k));
546  float c1 = b * (a + b) * (k - 1) / (a + b * (1 - k));
547 
548  endPts[i * 4 + 0][0] = pt0x + (pt0x - pt2x) / (a + b) * c0;
549  endPts[i * 4 + 0][1] = pt0y + (pt0y - pt2y) / (a + b) * c0;

893  / 2;

895  (locMat.at<float>(4, 0) - locMat.at<float>(3, 0))
896  * (locMat.at<float>(4, 0) - locMat.at<float>(3, 0))
897  + (locMat.at<float>(4, 1) - locMat.at<float>(3, 1))
898  * (locMat.at<float>(4, 1) - locMat.at<float>(3, 1)))
899  / 2;

902  float y_mid = locMat.at<float>(4, 1);
903 
904  // get the angle of the positive x axis

906  locMat.at<float>(4, 0) - locMat.at<float>(1, 0));

918  CL_MEM_READ_WRITE, n_check * sizeof(cl_uchar), NULL, &err);

949  int validy = -1;
